// Wave 11-A: free-path equivalent of process-payment-completed.
// Listens on registration/created, gates on paymentStatus === "FREE",
// uses notificationSentAt as an atomic claim (existing field, indexed).
// Mirrors sendNotificationWithAtomicClaim pattern at
// process-payment-completed-helpers.ts:156-212 so behavior under SMTP
// failure + Inngest retry is identical to the paid path.

export const handleRegistrationCreatedFree = inngest.createFunction(
  { id: "handle-registration-created-free", retries: 3 },
  { event: "registration/created" },
  async ({ event, step }) => {
    const { registrationId } = event.data;

    // Codex H3-2: master kill switch. Default off until verifications green.
    if (process.env.REGISTRATION_NOTIFICATION_REFACTOR_ENABLED !== "true") {
      return { ok: true, skipped: "feature_flag_disabled" };
    }
    // Codex H4-1: cutoff is MANDATORY when flag is on. Fail closed on missing/invalid.
    // Compare against Registration.createdAt (durable, indexed) NOT event.ts (Inngest replay
    // may pass current/no ts, so event.ts is fail-open).
    const cutoffRaw = process.env.REGISTRATION_HANDLER_CUTOFF_AT;
    if (!cutoffRaw) {
      console.error("[handle-registration-created-free] missing REGISTRATION_HANDLER_CUTOFF_AT — refusing to send");
      return { ok: true, skipped: "missing_cutoff_fail_closed" };
    }
    const cutoffMs = Date.parse(cutoffRaw);
    if (Number.isNaN(cutoffMs)) {
      console.error("[handle-registration-created-free] invalid REGISTRATION_HANDLER_CUTOFF_AT", { cutoffRaw });
      return { ok: true, skipped: "invalid_cutoff_fail_closed" };
    }

    return step.run("send-with-claim", async () => {
      const reg = await db.registration.findUnique({
        where: { id: registrationId },
        include: { workshop: { include: { coach: true } } },
      });
      if (!reg) return { ok: true, skipped: "not_found" };
      // H4-1: pre-cutoff guard uses Registration.createdAt — not event.ts.
      if (reg.createdAt.getTime() < cutoffMs) return { ok: true, skipped: "pre_cutoff" };
      if (reg.paymentStatus !== "FREE") return { ok: true, skipped: "paid_path_handles" };
      if (reg.notificationSentAt) return { ok: true, skipped: "already_sent" };

      const claimedAt = new Date();
      const claim = await db.registration.updateMany({
        where: { id: reg.id, notificationSentAt: null },
        data: { notificationSentAt: claimedAt },
      });
      if (claim.count === 0) return { ok: true, skipped: "race_lost" };

      try {
        const icsContent = generateIcsContent({
          uid: `workshop-${reg.workshop.id}@scaling-up-platform.com`,
          title: reg.workshop.title,
          description: reg.workshop.description,
          eventDate: reg.workshop.eventDate,
          eventTime: reg.workshop.eventTime,
          timezone: reg.workshop.timezone,
          durationHours: parseDurationHours(reg.workshop.duration),
          location: buildLocationString(reg.workshop),
        });
        const safeTitle = reg.workshop.title
          .replace(/[^a-zA-Z0-9-_ ]/g, "")
          .replace(/\s+/g, "-")
          .substring(0, 50);

        await sendRegistrationNotificationStrict({
          registrationId: reg.id, // Codex M3-1: telemetry hook
          workshopId: reg.workshop.id,
          workshopTitle: reg.workshop.title,
          workshopCode: reg.workshop.workshopCode,
          coachEmail: reg.workshop.coach.email,
          coachName: `${reg.workshop.coach.firstName} ${reg.workshop.coach.lastName}`,
          registrantName: `${reg.firstName} ${reg.lastName}`,
          registrantEmail: reg.email,
          registrantCompany: reg.company ?? undefined,
          icsAttachment: { filename: `${safeTitle}.ics`, content: icsContent },
        });
        return { ok: true, registrationId };
      } catch (err) {
        // Roll back the claim scoped to OUR claimedAt so Inngest retry can re-send.
        await db.registration.updateMany({
          where: { id: reg.id, notificationSentAt: claimedAt },
          data: { notificationSentAt: null },
        });
        throw err;
      }
    });
  },
);
```

/**
 * Parse a Workshop.eventTime range string into start + end {hours, minutes}.
 * Handles: "15:00 - 17:00", "15:00-17:00", "15:00" (no end → null), "" / null.
 * Returns null if unparseable.
 *
 * Validates per Codex M2-2 (Round 2):
 *  - hours in [0, 23]
 *  - minutes in [0, 59]
 *  - If both start and end are present, asserts end > start (returns null on
 *    end <= start to prevent negative-duration ICS payloads).
 */
export function parseEventTimeRange(eventTime: string | null | undefined): {
  startHours: number;
  startMinutes: number;
  endHours: number | null;  // null = caller must use durationHours fallback
  endMinutes: number | null;
} | null;
```

// If the draft predates the 2026-05-07 VIRTUAL-default flip AND has
// empty venue fields, flip to VIRTUAL on load. Belt-and-suspenders with
// the server-side cleanup script; this catches drafts loaded into an
// open wizard mid-script-run or saved-back after.
const cutoff = new Date("2026-05-07T00:00:00.000Z").getTime();

  draftCreatedAt < cutoff &&
  !parsedData.venueName &&

  !parsedData.venueCity &&
